171 class CTypeInfo {
172  public:
173  enum class Type : char {
174  kVoid,
175  kBool,
176  kInt32,
177  kUint32,
178  kInt64,
179  kUint64,
180  kFloat32,
181  kFloat64,
182  kUnwrappedApiObject,
183  };
184 
185  enum class ArgFlags : uint8_t {
186  kNone = 0,
187  kIsArrayBit = 1 << 0, // This argument is first in an array of values.
188  };
189 
190  static CTypeInfo FromWrapperType(const void* wrapper_type_info,
191  ArgFlags flags = ArgFlags::kNone) {
192  uintptr_t wrapper_type_info_ptr =
193  reinterpret_cast<uintptr_t>(wrapper_type_info);
194  // Check that the lower kIsWrapperTypeBit bits are 0's.
195  CHECK_EQ(
196  wrapper_type_info_ptr & ~(static_cast<uintptr_t>(~0)
197  << static_cast<uintptr_t>(kIsWrapperTypeBit)),
198  0u);
199  // TODO(mslekova): Refactor the manual bit manipulations to use
200  // PointerWithPayload instead.
201  return CTypeInfo(wrapper_type_info_ptr | static_cast<int>(flags) |
202  kIsWrapperTypeBit);
203  }
204 
205  static constexpr CTypeInfo FromCType(Type ctype,
206  ArgFlags flags = ArgFlags::kNone) {
207  // ctype cannot be Type::kUnwrappedApiObject.
208  return CTypeInfo(
209  ((static_cast<uintptr_t>(ctype) << kTypeOffset) & kTypeMask) |
210  static_cast<int>(flags));
211  }
212 
213  const void* GetWrapperInfo() const;
214 
215  constexpr Type GetType() const {
216  if (payload_ & kIsWrapperTypeBit) {
217  return Type::kUnwrappedApiObject;
218  }
219  return static_cast<Type>((payload_ & kTypeMask) >> kTypeOffset);
220  }
221 
222  constexpr bool IsArray() const {
223  return payload_ & static_cast<int>(ArgFlags::kIsArrayBit);
224  }
225 
226  private:
227  explicit constexpr CTypeInfo(uintptr_t payload) : payload_(payload) {}
228 
229  // That must be the last bit after ArgFlags.
230  static constexpr uintptr_t kIsWrapperTypeBit = 1 << 1;
231  static constexpr uintptr_t kWrapperTypeInfoMask = static_cast<uintptr_t>(~0)
232  << 2;
233 
234  static constexpr unsigned int kTypeOffset = kIsWrapperTypeBit;
235  static constexpr unsigned int kTypeSize = 8 - kTypeOffset;
236  static constexpr uintptr_t kTypeMask =
237  (~(static_cast<uintptr_t>(~0) << kTypeSize)) << kTypeOffset;
238 
239  const uintptr_t payload_;
240 };
